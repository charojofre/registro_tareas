📌 Paso 1: Crear Google Sheets

1. Ve a sheets.new y crea una hoja llamada "Registro_Tareas"

2. Compártela con "Cualquier persona con el enlace puede editar"

3. Copia el enlace del sheet

🔧 Paso 2: Configurar Apps Script

1. En tu Google Sheets, ve a Extensiones → Apps Script

2. Pega el código del servidor (está más abajo en esta página)

3. Haz clic en "Implementar" → "Nueva implementación"

4. Tipo: "Aplicación web", acceso: "Cualquier persona"

5. Copia la URL que aparece (termina en /exec)

Ejemplo: https://script.google.com/macros/s/ABC123/exec

📋 Código para Google Apps Script

Copia y pega este código en Extensiones → Apps Script en tu Google Sheets:

// ========== CONFIGURACIÓN ========== const SHEET_NAME = "Registro_Tareas"; function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); // Crear hoja si no existe if (!sheet) { const newSheet = SpreadsheetApp.getActiveSpreadsheet().insertSheet(SHEET_NAME); newSheet.appendRow(['Fecha', 'Hora', 'Persona', 'Proyecto', 'Descripción', 'ID', 'Estado', 'Timestamp']); return handleRequest(e); } // Verificar encabezados const firstRow = sheet.getRange(1, 1, 1, 8).getValues()[0]; if (!firstRow[0]) { sheet.appendRow(['Fecha', 'Hora', 'Persona', 'Proyecto', 'Descripción', 'ID', 'Estado', 'Timestamp']); } const params = e ? e.parameter : {}; const action = params.action; // Obtener todas las tareas if (action === "getTasks") { const data = sheet.getDataRange().getValues(); const tasks = []; for (let i = 1; i < data.length; i++) { if (data[i][0] && data[i][0] !== "") { tasks.push({ date: data[i][0], time: data[i][1], person: data[i][2], project: data[i][3], description: data[i][4], id: data[i][5], status: data[i][6] || 'pendiente', timestamp: data[i][7] }); } } return ContentService.createTextOutput(JSON.stringify({ success: true, tasks: tasks })).setMimeType(ContentService.MimeType.JSON); } // Agregar nueva tarea if (action === "addTask") { const newTask = { date: params.date || new Date().toISOString().split('T')[0], time: params.time || new Date().toLocaleTimeString('es-AR'), person: params.person, project: params.project || "Sin proyecto", description: params.description, id: Date.now().toString(), status: params.status || 'pendiente', timestamp: new Date().toISOString() }; sheet.appendRow([ newTask.date, newTask.time, newTask.person, newTask.project, newTask.description, newTask.id, newTask.status, newTask.timestamp ]); return ContentService.createTextOutput(JSON.stringify({ success: true, task: newTask })).setMimeType(ContentService.MimeType.JSON); } // Eliminar tarea if (action === "deleteTask") { const taskId = params.id; const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][5] === taskId) { sheet.deleteRow(i + 1); break; } } return ContentService.createTextOutput(JSON.stringify({ success: true, message: "Tarea eliminada" })).setMimeType(ContentService.MimeType.JSON); } // Actualizar estado if (action === "updateStatus") { const taskId = params.id; const newStatus = params.status; const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][5] === taskId) { sheet.getRange(i + 1, 7).setValue(newStatus); break; } } return ContentService.createTextOutput(JSON.stringify({ success: true, message: "Estado actualizado" })).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({ success: false, error: "Acción no válida" })).setMimeType(ContentService.MimeType.JSON); }
